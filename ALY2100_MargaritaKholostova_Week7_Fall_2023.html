"""
This program is designed to manage and store details about electronic items, specifically generic gadgets and laptops.

Features:
1. Abstract Base Class - 'ElectronicItem':
    - Represents a general electronic item with attributes like brand, model, price, and warranty period.
    - Allows displaying the item's information in a user-friendly format.

2. Parent Class - 'Gadget':
    - Derived from 'ElectronicItem'.
    - Represents a general gadget.
    - Allows setting and getting attributes (price, warranty, brand, model).
    - Provides functionality to save gadget information to a CSV file named 'data.csv'.

3. Child Class - 'Laptop':
    - Derived from 'Gadget'.
    - Represents a laptop with additional attributes like RAM, processor, and storage.
    - Allows setting and getting these new attributes.
    - Overrides the display function to include the laptop-specific details.
    - Provides functionality to save laptop information to a CSV file named 'laptop_data.csv'.

Utility Functions:
1. 'get_non_empty_input' - Repeatedly prompts the user for input until a non-empty response is provided.
2. 'display_file_content' - Displays the content of a specified file.
3. 'display_csv_as_table' - Displays the content of a CSV file in a tabular format.

Main Flow:
- The program prompts the user to enter details for either a gadget or a laptop.
- Once the user has entered all the details, they have the option to save this data to a CSV file.
- The user can continue to enter details for multiple items.
- On exiting (by typing 'q'), the program displays the saved details in a tabular format from the CSV files.

Note: The CSV file handling ensures that data is appended without overwriting previous entries, and appropriate headers are added if the file is new.
"""


from typing import Any, Optional
import os

#abstract class 
class ElectronicItem:
    def __init__(self, brand:str, model:str, price: Optional[float] = None,warranty_period: Optional[float] = None)-> None: #price & warranty are not required to be known at the moment of object creation (default value None)
        self.brand = brand
        self.model = model
        self.price = price #in dollars float
        self.warranty_period = warranty_period #in months

    def display_info(self) -> None:
        print("Here is the data that you entered:")
        print(f"Manufacturer: {self.brand}")
        print(f"Model Number: {self.model}")
        if self.price is not None:
            print(f"Retail Price: ${self.price:.2f}")
        else:
            print("Retail Price: Not provided")
        if self.warranty_period is not None:
            print(f"Warranty Period: {self.warranty_period} months")
        else:
            print("Warranty Period: Not provided")
        print("\n")




#parent class
class Gadget(ElectronicItem):

    def __init__(self, brand:str, model:str, price: Optional[float] = None,warranty_period: Optional[float] = None) -> None: 
        super().__init__( brand, model, price, warranty_period)

    def set_price(self, price: float) -> None:
        if price < 0:
            raise ValueError("Price cannot be negative")
        self.price = price

    
    def set_warranty(self, warranty:float) -> None:
        if warranty < 0:
            raise ValueError("Warranty period cannot be negative")
        self.warranty_period = warranty
    
    def get_brand(self)->str:
        return self.brand
    
    def get_model(self)->str:
        return self.model

    def get_price(self)->Optional[float]:
        return self.price
    
    def get_warranty_period(self)-> Optional[float]:
        return self.warranty_period
    
    def save_to_file(self):
        filename = 'data.csv'
        try:
            file_exist = os.path.isfile(filename) and os.path.getsize(filename) > 0  # check if file exists and is not empty

            with open(filename, 'a') as file: 
                if not file_exist:
                    file.write("Brand, Model, Price, Warranty (in month)\n")
            
                # write data
                file.write(f"{self.brand}, {self.model}, {self.price}, {self.warranty_period}\n")
        except IOError as e:
            print(f"Error saving to file {e}")
    
        return filename  # return the filename after saving
    


#child class
class Laptop(Gadget):


    def __init__(self, brand:str, model:str, RAM:int, processor:str, storage:int, price: Optional[float]= None,warranty_period: Optional[float] = None) -> None:
        super().__init__(brand, model, price, warranty_period) # no need to specify types again when calling parents method
        self.RAM = RAM
        self.processor = processor
        self.storage = storage

    #getter & setters for earlier not mentioned RAM and Storage

    def set_RAM(self, RAM) -> None:
        if RAM <= 0:
            raise ValueError("RAM cannot be negative or equal to zero")
        self.RAM = RAM

    def set_processor(self, processor) -> None:
        self.processor = processor
    
    def set_storage(self, storage) -> None:
        if storage <= 0:
            raise ValueError("Storage cannot be negative or equal to zero")
        self.storage = storage
    
    def get_RAM(self) -> int:
        return self.RAM
    
    #storage will be stored in GB
    def get_storage(self) ->int:
        return self.storage
    
    def get_processor(self) ->str:
        return self.processor
    
    def display_info(self)-> None:
        super().display_info()  # calling the display_info from the parent class to display common attributes
        print(f"RAM: {self.RAM} GB")
        print(f"Processor: {self.processor}")
        print(f"Storage: {self.storage} GB")
        print("\n")



    def save_to_file(self):
        filename = "laptop_data.csv"
        try:
            file_exists_laptop = os.path.isfile(filename) and os.path.getsize(filename) > 0

            with open(filename, 'a') as file:
                if not file_exists_laptop:
                    file.write("Brand,Model,RAM,Processor,Storage,Price,Warranty \n")
        
                file.write(f"{self.brand},{self.model},{self.RAM},{self.processor},{self.storage},{self.price},{self.warranty_period}\n")
        except IOError as e:
            print(f"Error saving to the file: {e}")

        return filename #return filename after saving


def get_non_empty_input(prompt: str) -> str:
    """Prompt the user for input until they provide a non-empty response."""
    while True:
        user_input = input(prompt).strip()
        if user_input:
            return user_input
        print("Input cannot be empty. Please try again.")


#the format of displaying in csv was chosen, but function stays in case of needed change
def display_file_content(filename: str):
    """Displays the content of a specified file."""
    try:
        with open(filename, 'r') as file:
            print(f"\nContents of {filename}:")
            print(file.read())
    except IOError as e:
        print(f"Error reading from file {filename}: {e}")

def display_csv_as_table(filename: str):
    """Displays the content of a specified file the table format."""
    try:
        with open(filename, 'r') as file:
            lines = file.readlines()
            
            # Display header
            headers = lines[0].strip().split(',')
            header_format = '|'.join([' {:<15} '] * len(headers))
            print(header_format.format(*headers))
            print('-' * (15 * len(headers) + len(headers) - 1))
            
            # Display rows
            for line in lines[1:]:
                row = line.strip().split(',')
                print(header_format.format(*row))
            
    except IOError as e:
        print(f"Error reading from file {filename}: {e}")
